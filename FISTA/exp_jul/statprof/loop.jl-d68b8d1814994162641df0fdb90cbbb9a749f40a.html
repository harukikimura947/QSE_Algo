99 If the default GLib main event loop is not already running, start a Julia task
100 that runs it. Returns the task. If `wait` is true, it will block until the

102
103 See also [`stop_main_loop`](@ref).
104 """
105 function start_main_loop(wait=false)
106 @debug("Starting GLib main loop using g_main_context_iteration()")
107 # if a glib main-loop is already running, we don't need to start a new one
108 if !is_loop_running()
109 g_main_running[] = true
110 global glib_main_task = schedule(Task(glib_main))
111 end
112 if wait
113 i=1
114 while !is_loop_running()
115 sleep(0.001)
116 i+=1
117 if i>1000
118 @warn("Main loop is not starting, returning from start_main_loop() anyway!")
119 break
120 end
121 end
122 end
123 glib_main_task
124 end
