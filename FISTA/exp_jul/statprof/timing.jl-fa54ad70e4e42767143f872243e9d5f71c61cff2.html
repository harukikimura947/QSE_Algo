46 gc_total_bytes(gc_num::GC_Num) =
47 gc_num.allocd + gc_num.deferred_alloc + gc_num.total_allocd

49 function GC_Diff(new::GC_Num, old::GC_Num)
50 # logic from `src/gc.c:jl_gc_total_bytes`
51 old_allocd = gc_total_bytes(old)
52 new_allocd = gc_total_bytes(new)
53 return GC_Diff(new_allocd - old_allocd,
54 new.malloc - old.malloc,
55 new.realloc - old.realloc,
56 new.poolalloc - old.poolalloc,
57 new.bigalloc - old.bigalloc,
58 new.freecall - old.freecall,
59 new.total_time - old.total_time,
60 new.pause - old.pause,
61 new.full_sweep - old.full_sweep)
62 end

64 function gc_alloc_count(diff::GC_Diff)
65 diff.malloc + diff.realloc + diff.poolalloc + diff.bigalloc
66 end

113 function prettyprint_getunits(value, numunits, factor)
114 if value == 0 || value == 1
115 return (value, 1)
116 end
117 unit = ceil(Int, log(value) / log(factor))
118 unit = min(numunits, unit)
119 number = value/factor^(unit-1)
120 return number, unit
121 end

139 function time_print(io::IO, elapsedtime, bytes=0, gctime=0, allocs=0, compile_time=0, recompile_time=0, newline=false; msg::Union{String,Nothing}=nothing)
140 timestr = Ryu.writefixed(Float64(elapsedtime/1e9), 6)
141 str = sprint() do io
142 if msg isa String
143 print(io, msg, ": ")
144 else
145 print(io, length(timestr) < 10 ? (" "^(10 - length(timestr))) : "")
146 end
147 print(io, timestr, " seconds")
148 parens = bytes != 0 || allocs != 0 || gctime > 0 || compile_time > 0
149 parens && print(io, " (")
150 if bytes != 0 || allocs != 0
151 allocs, ma = prettyprint_getunits(allocs, length(_cnt_units), Int64(1000))
152 if ma == 1
153 print(io, Int(allocs), _cnt_units[ma], allocs==1 ? " allocation: " : " allocations: ")
154 else
155 print(io, Ryu.writefixed(Float64(allocs), 2), _cnt_units[ma], " allocations: ")
156 end
157 print(io, format_bytes(bytes))
158 end
159 if gctime > 0
160 if bytes != 0 || allocs != 0
161 print(io, ", ")
162 end
163 print(io, Ryu.writefixed(Float64(100*gctime/elapsedtime), 2), "% gc time")
164 end
165 if compile_time > 0
166 if bytes != 0 || allocs != 0 || gctime > 0
167 print(io, ", ")
168 end
169 print(io, Ryu.writefixed(Float64(100*compile_time/elapsedtime), 2), "% compilation time")
170 end
171 if recompile_time > 0
172 perc = Float64(100 * recompile_time / compile_time)
173 # use "<1" to avoid the confusing UX of reporting 0% when it's >0%
174 print(io, ": ", perc < 1 ? "<1" : Ryu.writefixed(perc, 0), "% of which was recompilation")
175 end
176 parens && print(io, ")")
177 newline && print(io, "\n")
178 end
179 print(io, str)
180 nothing
181 end

183 function timev_print(elapsedtime, diff::GC_Diff, compile_times; msg::Union{String,Nothing}=nothing)
184 allocs = gc_alloc_count(diff)
185 compile_time = first(compile_times)
186 recompile_time = last(compile_times)
187 time_print(stdout, elapsedtime, diff.allocd, diff.total_time, allocs, compile_time, recompile_time, true; msg)
188 padded_nonzero_print(elapsedtime, "elapsed time (ns)")
189 padded_nonzero_print(diff.total_time, "gc time (ns)")
190 padded_nonzero_print(diff.allocd, "bytes allocated")
191 padded_nonzero_print(diff.poolalloc, "pool allocs")
192 padded_nonzero_print(diff.bigalloc, "non-pool GC allocs")
193 padded_nonzero_print(diff.malloc, "malloc() calls", false)
194 padded_nonzero_print(diff.realloc, "realloc() calls", false)
195 # always print number of frees if there are mallocs
196 padded_nonzero_print(diff.freecall, "free() calls", diff.malloc > 0)
197 minor_collects = diff.pause - diff.full_sweep
198 padded_nonzero_print(minor_collects, "minor collections")
199 padded_nonzero_print(diff.full_sweep, "full collections")
200 end

212 """
213 @time expr
214 @time "description" expr
215
216 A macro to execute an expression, printing the time it took to execute, the number of
217 allocations, and the total number of bytes its execution caused to be allocated, before
218 returning the value of the expression. Any time spent garbage collecting (gc), compiling
219 new code, or recompiling invalidated code is shown as a percentage.
220
221 Optionally provide a description string to print before the time report.
222
223 In some cases the system will look inside the `@time` expression and compile some of the
224 called code before execution of the top-level expression begins. When that happens, some
225 compilation time will not be counted. To include this time you can run `@time @eval ...`.
226
227 See also [`@showtime`](@ref), [`@timev`](@ref), [`@timed`](@ref), [`@elapsed`](@ref),
228 [`@allocated`](@ref), and [`@allocations`](@ref).
229
230 !!! note
231 For more serious benchmarking, consider the `@btime` macro from the BenchmarkTools.jl
232 package which among other things evaluates the function multiple times in order to
233 reduce noise.
234
235 !!! compat "Julia 1.8"
236 The option to add a description was introduced in Julia 1.8.
237
238 Recompilation time being shown separately from compilation time was introduced in Julia 1.8
239
240 ```julia-repl
241 julia> x = rand(10,10);
242
243 julia> @time x * x;
244 0.606588 seconds (2.19 M allocations: 116.555 MiB, 3.75% gc time, 99.94% compilation time)
245
246 julia> @time x * x;
247 0.000009 seconds (1 allocation: 896 bytes)
248
249 julia> @time begin
250 sleep(0.3)
251 1+1
252 end
253 0.301395 seconds (8 allocations: 336 bytes)
254 2
255
256 julia> @time "A one second sleep" sleep(1)
257 A one second sleep: 1.005750 seconds (5 allocations: 144 bytes)
258
259 julia> for loop in 1:3
260 @time loop sleep(1)
261 end
262 1: 1.006760 seconds (5 allocations: 144 bytes)
263 2: 1.001263 seconds (5 allocations: 144 bytes)
264 3: 1.003676 seconds (5 allocations: 144 bytes)
265 ```
